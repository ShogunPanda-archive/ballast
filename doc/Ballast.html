def self.in_em_thread(start_reactor = false, &block)
  if EM.reactor_running?
    run_in_thread(&block)
  elsif start_reactor
    EM.synchrony do
      Ballast.in_em_thread(&block)
      EM.stop
    end
  else
    block.call
  end
end
