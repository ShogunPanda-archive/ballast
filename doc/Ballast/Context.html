def method_missing(method, *args, &block)
  object = __getobj__

  if object[method] then
    object[method]
  else
    super(method, *args, &block)
  end
end

def self.build(owner, additional = {})
  super({
    owner: owner,
    errors: [],
    output: nil,
    response: HashWithIndifferentAccess.new
  }.merge(additional).ensure_access(:indifferent))
end
